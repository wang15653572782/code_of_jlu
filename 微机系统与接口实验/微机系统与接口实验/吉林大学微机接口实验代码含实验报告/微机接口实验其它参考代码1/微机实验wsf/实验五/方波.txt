CODE SEGMENT
	ASSUME CS:CODE
START:

	MOV CX,0FFH;循环次数

	;DA转换器接IOY0
L1:	MOV AL,00H
	MOV DX,0600H
	OUT DX,AL
	CALL DELAY
	MOV AL,0FFH
	OUT DX,AL
	CALL DELAY
	LOOP L1 

DELAY:
	PUSH CX
	MOV CX,0FFFFH
L3:	
	LOOP L3
		MOV CX,0FFFFH
L4:	
	LOOP L4
		MOV CX,0FFFFH
L5:	
	LOOP L5
	POP CX
	RET
CODE ENDS
	END START

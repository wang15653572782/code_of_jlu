		DW 128 DUP(?)

CODE SEGMENT
	ASSUME CS:CODE
START:
	MOV AL,90H       ;8255
	MOV DX,0606H
	OUT DX,AL

TT:	MOV DX,0640H
	OUT DX,AL
	CALL DELAY
	IN AL,DX
	MOV DX,0602H
	OUT DX,AL
	JMP TT
	INT 03H

DELAY PROC
	PUSH CX
	PUSH BX
	MOV CX,0FFH
	
ZZ2: MOV BX,0FFFH
ZZ1: DEC BX
	 JNZ ZZ1
	 LOOP ZZ2
	 POP BX
	 POP CX
	 RET
	 
	DELAY ENDP
CODE ENDS
END START

STACK SEGMENT STACK
    DW 32 DUP(?)
STACK ENDS
 
CODE SEGMENT
    ASSUME  CS:CODE, SS:STACK
 
START:

    MOV AX,0000H
    MOV DS,AX
    
    MOV CX, 02H       
JUCHI:
    MOV DX, 0600H     
    MOV AL, 00H       
JC1: 
    OUT DX, AL        
    CALL DELAY1       
    CMP AL,0FFH
    JE JC2
    INC AL            
    JMP JC1
JC2:
    LOOP JUCHI
 
    
    MOV CX, 02H       
JUXING:
    MOV DX, 0600H
    MOV AL, 00H       
    OUT DX, AL
    CALL DELAY2      
    MOV AL, 0FFH     
    OUT DX, AL
    CALL DELAY2       
       MOV AL, 00H       
    OUT DX, AL
    CALL DELAY2
    LOOP JUXING
 
  
    MOV CX, 02H     
SANJIAO:
SJ1:
    MOV DX, 0600H
    OUT DX, AL
    CALL DELAY1       
    CMP AL, 0FFH
    JE SJ2           
    INC AL            
    JMP SJ1
SJ2:
    MOV DX, 0600H
    OUT DX, AL
    CALL DELAY1       
    CMP AL, 00H
    JE SJ3
    DEC AL            
    JMP SJ2
SJ3:
    LOOP SANJIAO
 
    ;产生阶梯波
    MOV CX, 02H     
    MOV AX, 0FFH       

  
    MOV BL,05H         
    DIV BL             
    MOV BL, AL         
    MOV BH, 00H        
JIETI:
    MOV AX,0000H      
JT1:
    MOV DX, 0600H
    OUT DX, AL
    CMP AX, 00FFH      
    JAE JT2           
    CALL DELAY2        
    ADD AX, BX      
    JMP JT1
JT2:    
    LOOP JIETI
JMP START
 
DELAY1:                ;短延时
    PUSH CX
    MOV CX, 01FFH
D1: 
    PUSH AX
    POP AX
    LOOP D1
    POP CX
    RET
 
DELAY2:               ;长延时
    PUSH CX
    MOV CX, 0FFFFH
D2: 
    PUSH AX
    POP AX
    LOOP D2
    POP CX
    RET
 
CODE ENDS
    END START

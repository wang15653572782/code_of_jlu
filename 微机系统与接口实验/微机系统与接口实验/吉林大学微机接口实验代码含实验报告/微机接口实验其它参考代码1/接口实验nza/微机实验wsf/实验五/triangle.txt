CODE SEGMENT
	ASSUME CS:CODE
START:
	CLI
	MOV AL,0EFH
	OUT 21H,AL
	STI
	MOV CX,0AH
	;AD转换器接IOY3

	;DA转换器接IOY0
L1:	MOV AL,00H
	MOV DX,0600H
L2:	OUT DX,AL
	CALL DELAY
	INC AL
	JNZ L2      
	DEC AL
	DEC AL
L4:	OUT DX,AL
	CALL DELAY
	DEC AL
	JNZ L4
	LOOP L1 


DELAY:
	PUSH CX
	MOV CX,00FFH
L3:	
	LOOP L3
	POP CX
	RET
CODE ENDS
	END START

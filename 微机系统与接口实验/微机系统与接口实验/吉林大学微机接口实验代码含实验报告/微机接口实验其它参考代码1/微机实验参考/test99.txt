      ROL   AH,1	;按键不在此列，扫描下一列

      JMP   a1		;重新扫描

      JZ    a4		    ;按键在第1行
      ADD   BX,4	    ;按键在第2行

      ADD   BX,4	    ;按键在第3行

      JNZ   a5

      MOV   AL,CS:[DI+BX]
      MOV   CS:[SI],AL	;将新按键存入小地址

       AND   AL,0FH
       CMP   AL,0FH
       JNZ   a7

      PUSH  AX
      PUSH  BX
      PUSH  CX

      MOV   BX,0
      MOV   AH,0DFH ;1111 0111B   
Z1:   MOV   AL,AH
      MOV   DX,0600h
      OUT   DX,AL 
           
  	  MOV   AL,CS:[SI+BX]	   ;取数的段码
      MOV   DX,0602H
      OUT   DX,AL	   ;送段选码，b口

      MOV   CX,30H
      loop $
      MOV   AL,00H
      MOV   DX,0602h 
      OUT   DX,AL          ;消影
      CMP   AH,07FH         ;  01111111B
      JNZ   Z1
      POP   CX
      POP   BX
      POP   AX
      RET

DISPLAY  ENDP

DELAY PROC
      PUSH CX
      MOV  CX,0FFfH
      LOOP $
      POP  CX
      RET
DELAY ENDP

CODE SEGMENT
	ASSUME CS:CODE
START:
A1:                   ;方波
	MOV AL,00H
	MOV DX,0600H
	OUT DX,AL
	CALL AD
	CALL AD
	CALL AD	
	CALL AD
	CALL AD	  ;调用延时程序
	MOV AL,0FFH
	OUT DX,AL
	CALL AD
	CALL AD
	CALL AD
	CALL AD
	CALL AD
	JMP A1

AD:
	MOV CX,9000
S_D:
	PUSH AX
	POP AX
	LOOP S_D
	RET
CODE ENDS
	END START

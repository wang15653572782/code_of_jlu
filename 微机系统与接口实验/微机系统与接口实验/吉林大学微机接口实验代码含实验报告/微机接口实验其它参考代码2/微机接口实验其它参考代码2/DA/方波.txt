 CODE SEGMENT
   ASSUME CS:CODE 
START:
MOV DX,0600H
X:
MOV AL,001H 
OUT DX,AL
CALL DELAY
MOV AL,0FFH
OUT DX,AL
CALL DELAY
JZ X

DELAY: 
MOV CX,0FFFFH
LOOP1:
PUSH AX
POP AX
LOOP LOOP1
MOV CX,0FFFFH
LOOP2:
PUSH AX
POP AX
LOOP LOOP2
MOV CX,0FFFFH
LOOP3:
PUSH AX
POP AX
LOOP LOOP3
RET

CODE ENDS
END START

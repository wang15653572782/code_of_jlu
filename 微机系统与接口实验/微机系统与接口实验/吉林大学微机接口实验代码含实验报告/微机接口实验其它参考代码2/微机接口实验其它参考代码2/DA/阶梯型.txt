;TANGLE 
 CODE SEGMENT
   ASSUME CS:CODE 
START:
MOV DX,0600H
X:
MOV AL,00H
OUT DX,AL
CALL DELAY
MOV AL,04FH
OUT DX,AL
CALL DELAY
MOV AL,08FH
OUT DX,AL
CALL DELAY
MOV AL,0FFH
OUT DX,AL
CALL DELAY
JMP X

DELAY: 
MOV CX,0FFFFH
LOOP1:
PUSH BX
POP BX
LOOP LOOP1
MOV CX,0FFFFH
LOOP2:
PUSH BX
POP BX
LOOP LOOP2
MOV CX,0FFFFH
RET

CODE ENDS
END START
